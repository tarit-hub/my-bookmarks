สำรองข้อมูล Google Sheets

🛠️ ขั้นตอนเปิดการเชื่อมต่อ (ทำครั้งเดียวใน 1 นาที):
  1. ไปที่ Google Sheets แล้วสร้างสเปรดชีตว่างขึ้นมาใหม่
  2. กดเมนู ส่วนขยาย (Extensions) > Apps Script
  3. คัดลอกโค้ดสคริปต์ด้านล่างนี้ไปวางแทนที่ของเดิมทั้งหมด:
function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    
    sheet.clear();
    sheet.appendRow(["Title", "URL", "Category", "Tags", "Note", "Created At"]);
    
    var headerRange = sheet.getRange(1, 1, 1, 6);
    headerRange.setFontWeight("bold");
    headerRange.setBackground("#7c3aed");
    headerRange.setFontColor("#ffffff");
    
    if (data.bookmarks && data.bookmarks.length > 0) {
      for (var i = 0; i < data.bookmarks.length; i++) {
        var b = data.bookmarks[i];
        var tagsStr = b.tags ? b.tags.join(", ") : "";
        sheet.appendRow([b.title, b.url, b.category, tagsStr, b.note, b.createdAt]);
      }
    }
    
    return ContentService.createTextOutput(JSON.stringify({ status: "success" }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ status: "error", message: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
  1. กดปุ่ม ทำให้ใช้งานได้ (Deploy) > การทำให้ใช้งานได้ใหม่ (New deployment) ที่มุมขวาบน
  2. เลือกประเภทเป็น เว็บแอป (Web app) (กดที่เฟืองหากไม่เห็น)
  3. ตั้งค่าผู้มีสิทธิ์เข้าถึงเป็น ทุกคน (Anyone) แล้วกด Deploy
  4. คัดลอก URL เว็บแอป ที่ได้ มาวางในช่องกรอกด้านบนสุด และกดบันทึก!
